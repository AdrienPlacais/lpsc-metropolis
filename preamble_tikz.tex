\usepackage{tikzscale}
\usepackage{pgfplots}

\usetikzlibrary{fit}
\usetikzlibrary{shapes}
\usetikzlibrary{arrows.meta}
\usetikzlibrary{calc}

\usepgfplotslibrary{units}
\usepgfplotslibrary{colorbrewer}  % For ColorBrewer, a sweet colors generator
\usepgfplotslibrary{groupplots}   % For groups of plots (equivalent of subfig for pgf plots)

\pgfplotsset{%
   filter discard warning=false,
   enlarge x limits=false, 	% Prevent margins between first/last data and left/right axis.
   % x tick label style={/pgf/number format/.cd, set thousands separator={\,}}, % To avoid 1 000 written as 1,000
   % y tick label style={/pgf/number format/.cd, set thousands separator={\,}}, % To avoid 1 000 written as 1,000
   legend style={%
         fill opacity=0.8,
         draw opacity=1,
         text opacity=1,
         draw=white!80.0!black,
         font=\tiny,
      },
   every axis/.append style={
         label style={%
               font=\scriptsize,
            },
      },
   every tick label/.append style={%
         font=\scriptsize,
      },
   cycle list/Dark2-8, 			% Set the list of colors
   cycle multiindex* list={%	            % In order to also have different markers
         mark=*, mark=*, mark=*, mark=*\nextlist% List of markers
         Dark2-8\nextlist 		               % List of colors
      },
   % bar cycle list/Dark2-8,
   classic plot/.style={%
         width=\textwidth, height=.8*\textheight,
         grid=major,
         mark size=1pt,
      },
}
